  manifestVersion: 1;

    narrations: Array<{
      cacheKey: string;
      text: string;
      chapterTitle?: string;
      startMs: number;
      endMs: number;
      durationMs: number;
    }>;
  };
};

  mp4: { fileName: "video.mp4"; format: "mp4"; path: string };

  tourId: string;

  videos: PortableVideoOutputs;

};
