```typescript
export const PORTABLE_OUTPUT_MANIFEST_VERSION = 1;

export type PortableChecksum = {
  algorithm: "sha256";
  byteSize: number;
  hex: string;
};

export type PortableArtifactDescriptor = {
  path: string;
  mediaType: string;
  checksum: PortableChecksum;
};

export type PortableOutputManifest = {
  manifestVersion: 1;
  tour: { id: string; title: string };
  playback: { durationMs: number };
  artifacts: {
    videos: { mp4: PortableArtifactDescriptor; webm?: PortableArtifactDescriptor };
    poster: PortableArtifactDescriptor & { captureTimestampMs: number };
    captions: { srt: PortableArtifactDescriptor; vtt: PortableArtifactDescriptor };
    chapters: PortableArtifactDescriptor;
    audio: Array<PortableArtifactDescriptor & { cacheKey: string; durationMs: number }>;
  };
  timeline: {
    chapters: Array<{ title: string; startMs: number }>;
    narrations: Array<{
      cacheKey: string;
      text: string;
      chapterTitle?: string;
      startMs: number;
      endMs: number;
      durationMs: number;
    }>;
  };
};

export function parsePortableOutputManifest(value: unknown): PortableOutputManifest;
export function toPortableRelativePath(outputDir: string, filePath: string): string;
export function createPortableArtifactDescriptor(input: {
  outputDir: string;
  filePath: string;
  mediaType: string;
}): Promise<PortableArtifactDescriptor>;
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| output artifact paths -> manifest helpers | Local filesystem paths are turned into external contract references |
| output file bytes -> checksum descriptors | Final generated bytes become integrity metadata trusted by later ingest |
| untrusted JSON -> parse helper | Future consumers may read malformed `manifest.json` payloads |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01-01 | T | `packages/manifest/src/schema.ts` | mitigate | Use strict Zod objects with explicit required/optional fields so unknown keys and malformed shapes are rejected per D-01, D-02, and OUT-02. |
| T-05-01-02 | E | `packages/manifest/src/paths.ts` | mitigate | Reject empty, absolute, and escaping paths and normalize to output-root-relative POSIX paths only per D-03 and D-09. |
| T-05-01-03 | T | `packages/manifest/src/checksum.ts` | mitigate | Fix the checksum algorithm to SHA-256 and compute it from final file bytes so later phases cannot substitute heuristic or partial hashing per OUT-02. |
| T-05-01-04 | R | `packages/manifest/src/index.ts` | mitigate | Export one parse/checksum/path boundary from `@demohunter/manifest` so later generator code cannot silently fork the external contract per D-10. |
</threat_model>
